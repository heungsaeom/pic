#INCLUDE <16F877a.H>
#FUSES NOWDT, PUT, HS, NOPROTECT, NOLVP
#USE DELAY(CLOCK=16M)
INT8 DEM, BDN;
#int_timer1
void INTERRUPT_TIMER1()
{ 
BDN++;   
SET_TIMER1(15536);
IF (BDN == 10)   
{ 
BDN = 0;  OUTPUT_B(DEM); 
DEM = ~ DEM;
}
}
VOID MAIN()
{    
SET_TRIS_B(0x00);  
DEM = 0X00 ;   
OUTPUT_B(DEM);     
BDN= 0X00 ;
SETUP_TIMER_1(T1_INTERNAL | T1_DIV_BY_8);    
SET_TIMER1(15536);
ENABLE_INTERRUPTS(GLOBAL);    
ENABLE_INTERRUPTS(INT_TIMER1);
WHILE(true)
{ 

}
}
